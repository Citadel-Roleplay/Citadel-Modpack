term_length_laws = {
	default = term_length_long
	cumulative = no
	flag = admin_law

	term_length_short = {
		can_have = {
			government_allows = administrative
			has_realm_law = free_city_election_succession_law
			NOT = { agot_is_historical_free_city = yes }
		}
		can_pass = {
			trigger_if = {
				limit = {
					AND = {
						government_allows = administrative
						has_realm_law = free_city_election_succession_law
					}
				}
				custom_tooltip = {
					text = admin_law_cooldown_desc
					NOT = { has_variable = admin_law_cooldown }
				}
			}
		}
		on_pass = {
			## Set cooldown
			set_variable = {
				name = admin_law_cooldown
				years = 25
			}
		}
	}

franchise_laws = {
	default = franchise_lean_democratic
	cumulative = no
	flag = admin_law

	franchise_democratic_full = {
		can_keep = {
			government_allows = administrative
			trigger_if = {
				limit = {
					is_independent_ruler = no
				}
				liege = { has_realm_law = franchise_democratic_full }
			}
		}

		pass_cost = {prestige = {value = increase_crown_authority_prestige_cost multiply = 5}}

		can_pass = {
			trigger_if = {
				limit = {
					is_independent_ruler = no
					liege = { has_realm_law = franchise_democratic_full }
				}
				custom_tooltip = {
					text = imperial_bureaucracy_pending_law_change
					always = no
				}
			}
			trigger_if = {
				limit = {is_independent_ruler = yes}
				has_realm_law = franchise_lean_democratic
			}
		}

		modifier = {
			tyranny_gain_mult = 1
			vassal_opinion = -30
			county_opinion_add = 25
			monthly_county_control_growth_factor = 0.25
		}
	}
