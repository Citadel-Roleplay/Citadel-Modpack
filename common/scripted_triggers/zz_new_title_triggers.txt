

has_too_many_held_duchies_trigger = {
	highest_held_title_tier >= tier_kingdom
	trigger_if = {
		limit = {
			has_game_rule = more_than_two
		}
		any_held_title = {
			count >= 2
			title_counts_towards_too_many_duchies_trigger = yes
		}
	}
	trigger_if = {
		limit = {
			NOT = { has_game_rule = more_than_two }
		}
		any_held_title = {
			count >= 3
			title_counts_towards_too_many_duchies_trigger = yes
		}
	}
}
