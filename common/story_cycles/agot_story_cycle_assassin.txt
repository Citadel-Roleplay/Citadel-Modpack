story_divine_plot_assassin = {
	on_setup = {
		set_variable = {
			name = employer
			value = story_owner.employer
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	# Random events
	effect_group = {
		days = 30
		triggered_effect = {
			trigger = {
				OR = {
					NOT = { exists = story_owner.employer }
					NOT = { var:employer = story_owner.employer }
					story_owner.employer = {
						NOT = {
							any_scheme = {
								scheme_type = murder
							}
						}
					}
				}
			}
			effect = {
				story_owner = {
					save_scope_as = assassin
					employer ?= { save_scope_as = employer }
					move_to_pool = yes
					scope:employer ?= {
						send_interface_toast = {
							title = divine_plot_assassin_left
							custom_tooltip = divine_plot_assassin_left.tooltip
							left_icon = scope:assassin
						}
					}
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
}
