#Banking Custom Loc
agot_banking_interest_level = {
	type = character

	text = { # Low interest level
		trigger = { var:InterestLevel < 0 }
		localization_key = agot_bank_list_interest_low
	}

	text = { # High interest level
		trigger = { var:InterestLevel > 0 }
		localization_key = agot_bank_list_interest_high
	}

	text = { # Normal interest level
		trigger = { var:InterestLevel = 0 }
		localization_key = agot_bank_list_interest_normal
	}
}

agot_bank_name = {
	type = character

	text = {
		trigger = {
			NOT = {
				AND = {
					has_variable = IB_Shares
					var:IB_Shares > 0
				}
			}
			NOT = {
				AND = {
					has_variable = bank1_Shares
					var:bank1_Shares > 0
					global_var:bank1_founder.dynasty = title:d_nf_rogare.holder.dynasty
				}
			}
			NOT = {
				AND = {
					has_variable = bank1_Shares
					var:bank1_Shares > 0
					global_var:bank1_founder = { has_inactive_trait = ai_bank_founder }
				}
			}
			NOT = {
				AND = {
					has_variable = bank2_Shares
					var:bank2_Shares > 0
					global_var:bank2_founder = { has_inactive_trait = ai_bank_founder }
				}
			}
			NOT = {
				AND = {
					has_variable = bank3_Shares
					var:bank3_Shares > 0
					global_var:bank3_founder = { has_inactive_trait = ai_bank_founder }
				}
			}
		}
		localization_key = bank_name_normal
	}

	text = {
		trigger = {
			has_variable = IB_Shares
			var:IB_Shares > 0
		}
		localization_key = bank_name_default_IB
	}

	text = {
		trigger = {
			has_variable = bank1_Shares
			var:bank1_Shares > 0
			global_var:bank1_founder.dynasty = title:d_nf_rogare.holder.dynasty

			NOT = {
				global_var:bank1_founder = { has_inactive_trait = ai_bank_founder }
			}
		}
		localization_key = bank_name_default_Rogare
	}

	text = {
		trigger = {
			OR = {
				AND = {
					has_variable = bank1_Shares
					var:bank1_Shares > 0
					global_var:bank1_founder = { has_inactive_trait = ai_bank_founder }
				}
				AND = {
					has_variable = bank2_Shares
					var:bank2_Shares > 0
					global_var:bank2_founder = { has_inactive_trait = ai_bank_founder }
				}
				AND = {
					has_variable = bank3_Shares
					var:bank3_Shares > 0
					global_var:bank3_founder = { has_inactive_trait = ai_bank_founder }
				}
			}
		}
		localization_key = bank_name_default
	}
}

agot_bank_name_list = {
	type = character

	text = {
		trigger = {
			NOT = { has_character_flag = IB_Director }
			NOT = { title:d_nf_rogare.holder = this }
		}
		localization_key = bank_name_normal_list
	}

	text = {
		trigger = { has_character_flag = IB_Director }
		localization_key = bank_name_default_IB
	}

	text = {
		trigger = { title:d_nf_rogare.holder = this }
		localization_key = bank_name_default_Rogare
	}
}
