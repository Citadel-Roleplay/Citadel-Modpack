knight_tree_count = {
	value = 0
	every_in_global_list = {
		variable = known_knight_trees
		add = 1
	}
}

knight_tree_teacher_count = {
	value = 0
	every_in_global_list = {
		variable = known_knight_trees
		every_in_list = {
			variable = teacher_story_container
			every_in_list = {
				variable = student_container
				add = 1
			}
		}
	}
}

knight_tree_members_count = {
	value = 1
	var:knight_tree_being_shown = {
		every_in_list = {
			variable = teacher_story_container
			every_in_list = {
				variable = student_container
				add = 1
			}
		}
	}
}

knight_tree_members_alive_count = {
	value = 0
	var:knight_tree_being_shown = {
		every_in_list = {
			variable = teacher_story_container
			every_in_list = {
				variable = student_container
				limit = { is_alive = yes }
				add = 1
			}
		}
		if = {
			limit = { var:knight_tree_founder = { is_alive = yes } }
			add = 1
		}
	}
}

knight_trees_destroyed_count = {
	value = 0
	if = {
		limit = { exists = global_var:knight_trees_destroyed }
		add = global_var:knight_trees_destroyed
	}
}
