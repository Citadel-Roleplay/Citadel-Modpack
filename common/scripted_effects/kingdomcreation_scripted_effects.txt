
duskendale_creation_decision_effects = {
	# add de jure territory, grant kingdom
	
	title:k_duskendale = { set_capital_county = title:c_duskendale }

	title:d_duskendale = { add_to_temporary_list = duskendale_historical_list }
	title:d_stokeworth = { add_to_temporary_list = duskendale_historical_list }
	title:d_deerfield = { add_to_temporary_list = duskendale_historical_list }
	
	
	hidden_effect = {
		every_in_list = {
			list = duskendale_historical_list
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_duchy }
				set_de_jure_liege_title = title:k_duskendale
			}
		}
	}
	get_title = title:k_duskendale
	set_primary_title_to = title:k_duskendale
	if = {
		limit = { top_liege = { has_title = title:e_the_iron_throne } }
		title:k_duskendale = { set_de_jure_liege_title = title:e_the_iron_throne }
	}
	
	else_if = {
		limit = { is_independent_ruler = yes }
		title:k_duskendale = { set_de_jure_liege_title = title:e_the_iron_throne }
	}
	else = {}
	
}
