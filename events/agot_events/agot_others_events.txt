namespace = agot_others_events

#Gain wights after winning battles
agot_others_events.9000 = {
	hidden = yes
	scope = none

	trigger = {
		exists = side_primary_participant
		side_primary_participant.faith = { has_doctrine_parameter = raise_wights_from_dead }
		always = no #has_trait = other
	}

	immediate = {
		combat.location = {
			save_scope_as = combat_location
		}

		side_primary_participant = {
			spawn_army = {
				name = wight_army_name
				levies = 0
				men_at_arms = {
					type = wights
					stacks = 5
				}
				location = scope:combat_location
			}
		}
	}
}
